SELECT queries:


SELECT COUNT(`id`)
FROM `students`
WHERE `date_of_birth` >= '1990-01-01' AND `date_of_birth` <= '1990-12-31'


SELECT COUNT(`id`)
FROM `courses`
WHERE `cfu` > '10'


SELECT COUNT(`id`)
FROM `students`

SELECT COUNT(`id`)
FROM `courses`
WHERE `year` = '1' AND `period` = 'I semestre'
